// Problem: 3513. Number of Unique XOR Triplets I
// Link: https://leetcode.com/problems/number-of-unique-xor-triplets-i

class Solution:
    def uniqueXorTriplets(self, nums: List[int]) -> int:
        n = len(nums)
        if n <= 2:
            return n
        ans = 1
        while ans <= n:
            ans <<= 1
        return ans
